## Usage

```ts

function Example(){
    return (
        <MyButton
        text: "MyButton primary",
        txtcolor: 'black',
        bgcolor: 'blue',
        bordercolor: 'black',
        borderwidth: 2,
        disabled: false,
        hovertxtcolor: "white",
        hoverbgcolor: "yellow",

        />

};
```
